Features:
  - Create item
  - Read all items
  - Update item
  - Delete item
  - and more..

Installation:

    # Clone repository

    # Create a virtualenv(optional)
      python3 -m venv env


    # Install all dependencies
       pip install -r requirements.txt


    # Activate the virtualenv
      source venv/bin/activate or .venv/bin/activate

    # Run application
     ./manage.py runserver or python manage.py runserver
